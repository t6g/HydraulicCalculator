\begin{equation}
A = \frac{1}{2} (z_1 + z_2) y^2 + by
\end{equation}

\begin{equation}
P = \left(\sqrt{1+z_1^2} + \sqrt{1+z_2^2}\right)y + b
\end{equation}

\begin{equation}
\frac{\partial A}{\partial y} = (z_1 + z_2)y + b
\end{equation}

\begin{equation}
\frac{\partial ^2A}{\partial y^2} = z_1 + z_2
\end{equation}

\begin{equation}
\frac{dP}{dy} = \sqrt{1+z_1^2} + \sqrt{1+z_2^2}
\end{equation}
