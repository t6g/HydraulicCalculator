\begin{equation}
A = by
\end{equation}

\begin{equation}
P = b + 2y
\end{equation}

\begin{equation}
R_h = \frac{by}{b+2y}
\end{equation}

\begin{equation}
\frac{\partial A}{\partial y} = b = T_w
\end{equation}

\begin{equation}  
f_c(y)= gb^3y^3 -Q^2b= 0
\end{equation}

\begin{equation}  
y_c^3 = \frac{Q^2}{gb^2}
\label{Eq:RectCritical}
\end{equation}

\begin{equation}  
D_h = \frac{A}{\frac{\partial A}{\partial y}} = \frac{by_c}{b} = y_c
\end{equation}

\begin{equation}  
v_c = \sqrt{g y_c}
\end{equation}

\noindent Analytical solution is derived for critical depth (Eq: \ref{Eq:RectCritical}). Neton's method is used to calculate normal depth $y_n$.
